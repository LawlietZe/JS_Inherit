function curry () {
  // 把aruments中类数组转化成数组
 let argsList = [...arguments]
 const fn = function() {
  argsList.push(...arguments)
  // 递归把参数传递进来
  return fn
 }
 // toString是函数在编译器中最后会自动执行
 fn.toString = function() {
   return argsList.reduce((a,b)=> a+b)
 }
 return fn
}
